procedure TForm1.Button1Click(Sender: TObject); 
var a,b,c,d,s:integer; 
i:array[0..4,0..4] of integer; //ћассив, в котором все происходит 
begin 
Randomize; // √енератор случайных чисел 
Memo1.Clear; 
c:=strtoint (Edit1.Text); 
d:=0; 
s:=0; 
for a:=0 to 4 do begin //—тандартный цикл по всему массиву 
for b:=0 to 4 do begin 
i[a,b] := 10-random (20)+1; 
StringGrid1.Cells[a,b]:= IntToStr(i[a,b]); //«аполнение массива€ случайными числами 
end; 
end; 
for a:=0 to 4 do begin 
for b:=0 to 4 do begin 
if (i[a,b]<=10) and (i[a,b]>= 0) then begin //≈сли в массиве находитс€ положительное число меньше 10, то оно идет в сумму 
d:=d+i[a,b]; 
end; 
end; 
end; 
Memo1.Lines.Add('сумма положительных чисел меньше 10'); 
Memo1.Lines.Add(IntToStr(d)); 
for a:=0 to 4 do begin 
for b:=0 to 4 do begin 
if (i[a,b] >= c) and (i[a,b] mod 5 = 0) then begin //счетчик, который считает, сколько чисел кратные 5 и больше числа, которое написал пользователь 
s:=s+1; 
end; 
end; 
end; 
Memo1.Lines.Add('„исла кратные 5 и Ѕольше B');

end;
